    def get(self, url: object, timeout: object, proxy: object = None, num_retries: object = 6) -> object:  ##给函数一个默认参数proxy为空

        headers = {'Content-Type': 'utf-8'}
        headers['referer'] = url
        if proxy == None:  ##当代理为空时，不使用代理获取response（别忘了response啥哦！之前说过了！！）

                response = requests.get(url, headers=headers, timeout=timeout)  ##这样服务器就会以为我们是真的浏览器了

                    return response

                    IP = ''.join(str(random.choice(self.iplist)).strip())  ##下面有解释哦

                    return self.get(url, timeout, proxy, )
